--Student Table
CREATE TABLE `college`.`student` (`ADMISSION_NUMBER` VARCHAR(25) NOT NULL , `STUDENT_ID` VARCHAR(25) NOT NULL , `STUDENT_NAME` VARCHAR(25) NOT NULL , `DOB` DATE NOT NULL , `YEAR_OF_STUDY` INT NOT NULL , `YEAR_OF_JOINING` INT NOT NULL , `TENTH_GRADE_MARK` INT NOT NULL , `TWELFTH_GRADE_MARK` INT NOT NULL , `FIRST_GRADUATE` VARCHAR(20) NOT NULL , `EMAIL_ID` VARCHAR(25) NOT NULL , `PHONE_NUMBER` BIGINT(20) NOT NULL , PRIMARY KEY (`STUDENT_ID`(25))) ENGINE = InnoDB;

--Course Table
CREATE TABLE `college`.`course` (`COURSE_ID` VARCHAR(25) NOT NULL , `COURSE_NAME` VARCHAR(25) NOT NULL , `COURSE_COORDINATOR` VARCHAR(30) NOT NULL , `DEPARTMENT` VARCHAR(50) NOT NULL , `COURSE_FEE` DOUBLE NOT NULL , PRIMARY KEY (`COURSE_ID`(25))) ENGINE = InnoDB;	

--Enrollment Table
create table enrollment(ENROLLMENT_ID VARCHAR(20) PRIMARY KEY,
	STUDENT_ID VARCHAR(20),
	COURSE_ID VARCHAR(20),
	FEE_STATUS VARCHAR(20),
	FOREIGN KEY(STUDENT_ID) REFERENCES STUDENT(STUDENT_ID),
	FOREIGN KEY(COURSE_ID) REFERENCES COURSE(COURSE_ID));
--Attendance Table

create table attendance( ATTENDANCE_ID VARCHAR(30) PRIMARY KEY, STUDENT_ID VARCHAR(30), ENROLLMENT_ID VARCHAR(30), SEMESTER INT, TOTAL_WORKING_DAYS INT, PRESENT_DAYS INT, ABSENT_DAYS INT, ATTENDANCE_PERCENTAGE INT, FOREIGN KEY(STUDENT_ID) REFERENCES student(STUDENT_ID), FOREIGN KEY(ENROLLMENT_ID) REFERENCES enrollment(ENROLLMENT_ID));

--Mark Table
create table mark( MARK_ID VARCHAR(20) PRIMARY KEY, STUDENT_ID VARCHAR(25), SEMESTER INT, SUBJECT1 VARCHAR(40), SUBJECT2 VARCHAR(40), SUBJECT3 VARCHAR(40), SUBJECT4 VARCHAR(40), SUBJECT5 VARCHAR(40), SUBJECT6 VARCHAR(40), SUBJECT7 VARCHAR(40), SUBJECT8 VARCHAR(40), GPA DOUBLE, CGPA DOUBLE, FOREIGN KEY(STUDENT_ID) REFERENCES student(STUDENT_ID));

--Payment Table
create table payment(
    PAYMENT_ID VARCHAR(30) PRIMARY KEY,
      ENROLLMENT_ID VARCHAR(30),
    PAYMENT_DATE DATE,
    PAYMENT_MODE VARCHAR(30),
      AMOUNT DOUBLE,
    FOREIGN KEY(ENROLLMENT_ID) REFERENCES enrollment(ENROLLMENT_ID));
